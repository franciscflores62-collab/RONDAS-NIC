    const SESSION_KEY = "gsd_current_user";

    const viewer = document.getElementById("viewer");
    const siteName = document.getElementById("siteName");
    const formFrame = document.getElementById("formFrame");
    const frameLoading = document.getElementById("frameLoading");

    const authModal = document.getElementById("authModal");

    const toastWrap = document.getElementById("toastWrap");

    year.textContent = new Date().getFullYear();

    function normalizarUsuario(valor) {
      return valor.trim().toUpperCase();
    }

    function obtenerFormulario(formId) {
      return FORMULARIOS.find(item => item.id === formId) || null;
    }

    function obtenerUsuarioSesion() {
      const key = sessionStorage.getItem(SESSION_KEY);
      if (!key || !USUARIOS[key]) return null;
      return { key, ...USUARIOS[key] };
    }

    function guardarSesion(userKey) {
      sessionStorage.setItem(SESSION_KEY, userKey);
    }

    function limpiarSesion() {
      sessionStorage.removeItem(SESSION_KEY);
    }

    function mostrarToast(mensaje, tipo = "info") {
      const toast = document.createElement("div");
      toast.className = `toast ${tipo}`;
      toast.textContent = mensaje;
      toastWrap.appendChild(toast);

      setTimeout(() => {
        toast.style.opacity = "0";
        toast.style.transform = "translateY(8px)";
      }, 2800);

      setTimeout(() => {
        toast.remove();
      }, 3200);
    }

    function mostrarLogin() {
      authModal.style.display = "flex";
      authModal.setAttribute("aria-hidden", "false");
      document.body.classList.add("modal-open");
      viewer.classList.remove("active");
      viewer.setAttribute("aria-hidden", "true");
      document.body.classList.remove("viewer-open");
      formFrame.src = "";
      openExternalBtn.href = "#";
      frameLoading.classList.remove("hidden");
      loginError.style.display = "none";
      loginError.textContent = "";
      loginForm.reset();

      setTimeout(() => usernameInput.focus(), 50);
    }

    function ocultarLogin() {
      authModal.style.display = "none";
      authModal.setAttribute("aria-hidden", "true");
      document.body.classList.remove("modal-open");
    }

    function validarLogin(usuarioInput, passwordInputValue) {
      const userKey = normalizarUsuario(usuarioInput);
      const user = USUARIOS[userKey];

      if (!user) {
        return { ok: false, mensaje: "Usuario no válido." };
      }

      if (user.password !== passwordInputValue) {
        return { ok: false, mensaje: "Contraseña incorrecta." };
      }

      if (!user.formularios || user.formularios.length === 0) {
        return { ok: false, mensaje: "Este usuario no tiene formularios asignados." };
      }

      const formularioAsignado = obtenerFormulario(user.formularios[0]);

      if (!formularioAsignado) {
        return { ok: false, mensaje: "No se encontró el formulario asignado a este usuario." };
      }

      return {
        ok: true,
        userKey,
        user,
        formulario: formularioAsignado
      };
    }

    function abrirFormulario(form, user) {
      siteName.textContent = `${form.nombre} | ${user.nombre}`;
      openExternalBtn.href = form.url;
      frameLoading.classList.remove("hidden");
      formFrame.src = form.url;

      viewer.classList.add("active");
      viewer.setAttribute("aria-hidden", "false");
      document.body.classList.add("viewer-open");
    }

    function cerrarSesion() {
      limpiarSesion();
      mostrarLogin();
      mostrarToast("Sesión cerrada correctamente.", "info");
    }

    formFrame.addEventListener("load", () => {
      frameLoading.classList.add("hidden");
    });

    loginForm.addEventListener("submit", (e) => {
      e.preventDefault();

      const resultado = validarLogin(
        usernameInput.value,
        passwordInput.value
      );

      if (!resultado.ok) {
        loginError.textContent = resultado.mensaje;
        loginError.style.display = "block";
        mostrarToast(resultado.mensaje, "error");
        return;
      }

      guardarSesion(resultado.userKey);
      ocultarLogin();
      abrirFormulario(resultado.formulario, resultado.user);
      mostrarToast(`Bienvenido, ${resultado.user.nombre}.`, "success");
    });

    logoutViewerBtn.addEventListener("click", cerrarSesion);

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && viewer.classList.contains("active")) {
        cerrarSesion();
      }
    });

    function iniciarAplicacion() {
      const sesion = obtenerUsuarioSesion();

      if (!sesion) {
        mostrarLogin();
        return;
      }

      const formularioAsignado = obtenerFormulario(sesion.formularios[0]);

      if (!formularioAsignado) {
        limpiarSesion();
        mostrarLogin();
        mostrarToast("No se encontró el formulario asignado. Inicia sesión nuevamente.", "error");
        return;
      }

      ocultarLogin();
      abrirFormulario(formularioAsignado, sesion);
    }

    iniciarAplicacion();
